    function countdown( parentElementID, endDate, gameType, isFreeCountdown )
    {
        let element, endTime, hours, mins, msLeft, time, parentElement, timerDiv, day;
        let now = new Date();
        let seconds = (endDate.getTime() - now.getTime()) / 1000;
        let timerEndText = "";

        function twoDigits( n )
        {
            return (n <= 9 ? "0" + n : n);
        }

        function updateTimer()
        {
            let today = new Date();
            msLeft = endTime - (today.getTime());

            if( gameType == "jeux" && !isFreeCountdown && msLeft < 300000 ){
                timerDiv.innerHTML = "Moins de 5min pour jouer !";
            } else if ( msLeft < 1000 ) {
                if(timerEndText !== ""){
                    timerDiv.innerHTML = "";
                }
            } else {
                time = new Date( msLeft );
                hours = time.getUTCHours();
                if(time.getUTCSeconds() !== 0){
                    day = getdayDiff(new Date(), endDate);
                }
                mins = time.getUTCMinutes();
                element.innerHTML = (day ? twoDigits(day) + 'j ': '') + (day || hours ? twoDigits( hours ) + 'h ' + twoDigits( mins ) : mins) + 'm ' + twoDigits( time.getUTCSeconds()) + 's';

                setTimeout( updateTimer, time.getUTCMilliseconds() + 500 );
            }
        }

        function getdayDiff(date1, date2){
            let diff = {};
            let tmp = date2 - date1;

            tmp = Math.floor(tmp/1000);
            diff.sec = tmp % 60;

            tmp = Math.floor((tmp-diff.sec)/60);
            diff.min = tmp % 60;

            tmp = Math.floor((tmp-diff.min)/60);
            diff.hour = tmp % 24;

            tmp = Math.floor((tmp-diff.hour)/24);
            return tmp;
        }

        parentElement = document.getElementById( parentElementID );
        timerDiv = parentElement.querySelector('div.banner-timer');
        element = parentElement.querySelector('span.banner-timer_time');
        endTime = (now.getTime()) + 1000 * seconds + 500;
        updateTimer();
    }

    countdown( "a2614982-883d-45bc-84ac-09b6278ea405" , new Date(1602267300000), "jeux", false );
